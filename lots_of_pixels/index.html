          let g = 10, o = 0;

          function setup() {
            createCanvas(windowWidth, windowHeight);
            noStroke();
          }
          
          function draw() {
            o += 0.05;
            
            for (let i = 0; i < width; i += g) {
              for (let j = 0; j < height; j += g) {
                let d = dist(width / 2, height / 2, i + g / 2, j + g / 2);
                let a = map(d, 0, width / 2, 0, PI);
                let s = abs(sin(a * (map(mouseX, 0, width, 1, 155)) + o));
                
                fill(s > 0.5 ? 0 : width);
                rect(i, j, g, g);
              }
            }
          }
